Reference delegated JavaScript logic:

```js
function installLiquidControls() {
  const reduceMotion = window.matchMedia(
    "(prefers-reduced-motion: reduce)"
  );

  const release = (control) => {
    control.classList.remove("is-liquid-pressed");
  };

  const onPointerDown = (event) => {
    if (!(event.target instanceof Element)) return;

    const control = event.target.closest(
      ".liquid-control, button, [role='button']"
    );

    if (
      !control ||
      control.matches(":disabled") ||
      control.getAttribute("aria-disabled") === "true" ||
      reduceMotion.matches
    ) {
      return;
    }

    const rect = control.getBoundingClientRect();
    const diameter = Math.ceil(
      Math.sqrt(rect.width ** 2 + rect.height ** 2) * 2
    );
    const ripple = document.createElement("span");

    control.querySelectorAll(".liquid-ripple").forEach((node) => {
      node.remove();
    });

    ripple.className = "liquid-ripple";
    ripple.style.setProperty("--liquid-x", `${event.clientX - rect.left}px`);
    ripple.style.setProperty("--liquid-y", `${event.clientY - rect.top}px`);
    ripple.style.setProperty("--liquid-size", `${diameter}px`);

    control.appendChild(ripple);
    control.classList.add("is-liquid-pressed");

    const releasePress = () => release(control);

    window.addEventListener("pointerup", releasePress, { once: true });
    window.addEventListener("pointercancel", releasePress, { once: true });

    ripple.addEventListener(
      "animationend",
      () => ripple.remove(),
      { once: true }
    );
  };

  document.addEventListener("pointerdown", onPointerDown);

  return () => {
    document.removeEventListener("pointerdown", onPointerDown);
  };
}
```
